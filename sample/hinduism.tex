	\begin{enumerate}
		\item<1-13,15->\textbf{Br{\'a}hman}			\only<2-13,15->{: universal Soul}
		\item<3-13,15->\textbf{\={A}tman}			\only<4-13,15->{: one's self (or ``soul'')}
		\item<5-13,15->Time							\only<6-13,15->{: circular and repetitive}
		\item<7-13,15->\textbf{Karma}				\only<8-13,15->{: value of intentions behind actions}
			\begin{itemize}
				\item<9-13,15->Both ``good'' and ``bad'' karma are bad
				\item<10-13,15->Born with it
				\item<11-13,15->``Holiness'' = less karma
			\end{itemize}
		\item<12-13,15->\textbf{Sa\d{m}s{\=a}ra}	\only<13,15->{: cycle of death and rebirth}
			\begin{itemize}
				\item<15->Caused by karma
				\item<16->Gods are part of the cycle too
			\end{itemize}
		\item<17->\textbf{Mok\d{s}a}				\only<18->{: liberation from sa\d{m}s{\=a}ra}
			\begin{itemize}
				\item<19->Dvaita					\only<20->{ -- loving union with Br{\'a}hman}
				\item<21->Advaita					\only<22->{ -- realization of true nature of \={a}tman and Br{\'a}hman}
			\end{itemize}
	\end{enumerate}
\end{frame}
