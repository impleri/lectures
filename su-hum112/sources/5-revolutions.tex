\subsection{Vienna}
\begin{frame}{German Music}
	\begin{itemize}
		\item<1->Cantata\only<1>{: Multimovement choral adaptation of a text accompanied by music (comes from Latin meaning "to sing")}
		\item<2->Oratorio\only<2>{: Choral work without movement or scenery accompanied by music (e.g. Bach's \emph{Christmas Oratorio} [1734])}
		\item<3->Fugue\only<3>{: Progression of a single theme across multiple lines until a harmony is reached (e.g. Bach's \emph{Toccata and Fugue in Dm} [1740])}
		\item<4->Sonata\only<4>{: Solo musical piece without vocal accompaniment, either as a single instrument or a group (comes from Latin meaning "to sound")}
		\item<5->Orchestra\only<5>{: Largely organized ensemble of musicians, sometimes grouped by instruments (e.g. strings, reeds, etc) and led by a conductor}
		\item<6->Symphony\only<6>{: A three or four movement composition most frequently played by an orchestra}
		\item<7->Classical [1760--18??]
		\item<8->J. S. Bach [1685--1750]. (\emph{Brandenburg Concerto} [1721])
		\item<9->Joseph Haydn [1732--1809] (\emph{Symphony No. 94} [1791])
		\item<10->Wolfgang Amadeus Mozart [1756--1791] (\emph{Symphony no. 40} [1788], \emph{Marriage of Figaro} [1786])
	\end{itemize}
\end{frame}
