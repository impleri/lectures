\subsection{Colonizing the Americas}
\begin{frame}{Early Colonies}
	\slide{10}{Early settlements in America}{map-1700-north_america.pdf}
	\begin{itemize}
		\item<1-9>Incan Empire
		\item<2-9>Spain and Portugal were first
		\item<3-9>England and France did not agree to the Spanish/Portuguese treaty
		\item<4-9>First English and French attempts failed
			\only<4>{
				\begin{enumerate}
					\item 1534. France in Gasp{\'e} Peninsula (Quebec/New Brunswick north of Maine)
					\item 1564. French Hugenots in Fort Caroline (now Jacksonville, FL), sacked by Spanish
					\item 1586. British in Roanoke (Lost Colony)
					\item 1590s--1613. French in Acadia, multiple failures
					\item 1610--1637. English in Newfoundland
					\item 1629--1632. Scottish in Nova Scotia
				\end{enumerate}
			}
		\item<5-9>1606. English settle Virginia
		\item<6-9>1608. French settle city of Quebec
		\item<7-9>1614. Dutch settle New York, New Jersey, Connecticut
		\item<8-9>1620. English Puritans in Massachusetts
		\item<9>1638. Swedish in Delaware Valley (conquered by Dutch in 1655)
	\end{itemize}
\end{frame}
